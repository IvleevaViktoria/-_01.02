//Выполнить возведение заданной матрицы в заданную степень. Перемножение произвести поэлементно, а не по известным правилам перемножения из курса Высшей математики.
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp04
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Введите размерность квадратной матрицы: ");
            int n = Convert.ToInt32(Console.ReadLine());

            // Создание и заполнение исходной матрицы
            int[,] A = new int[n, n];
            Console.WriteLine("Введите элементы матрицы A:");
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    A[i, j] = Convert.ToInt32(Console.ReadLine());
                }
            }

            Console.WriteLine("Введите степень, в которую нужно возвести матрицу: ");
            int k = Convert.ToInt32(Console.ReadLine());

            // Возведение в степень
            int[,] result = A;
            for (int p = 2; p <= k; p++)
            {
                int[,] temp = new int[n, n];
                for (int i = 0; i < n; i++)
                {
                    for (int j = 0; j < n; j++)
                    {
                        // Перемножение произвести поэлементно
                        for (int m = 0; m < n; m++)
                        {
                            temp[i, j] += result[i, m] * A[m, j];
                        }
                    }
                }

                result = temp;
            }

            // Вывод результата
            Console.WriteLine("Матрица A в степени {0}:", k);
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    Console.WriteLine(result[i, j] + "\t");
                }

                Console.WriteLine();
            }

            Console.ReadLine();
        }
    }
}
