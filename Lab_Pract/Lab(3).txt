using System;
class HelloWorld {
  static void Main() 
  {
    double x,y,I = 0;
    int d,i=0;
    x = Convert.ToDouble(Console.ReadLine());
    y = Convert.ToDouble(Console.ReadLine());
    Console.WriteLine("Ogr");
    d = Convert.ToInt32(Console.ReadLine());
    
    for (i = 1; i <=d; i++)
    {
        I+=Convert.ToDouble((Math.Pow(0.1,i)*x + Math.Pow(Math.Sin(y),i)/Math.Sin(Math.Pow(x,i))));
        Console.WriteLine(I);
        
    }
    
  }
}
